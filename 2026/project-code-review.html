def get_user(user_id):
  import sqlite3
  conn = sqlite3.connect("users.db")
  cur = conn.cursor()
  query = f"SELECT * FROM users WHERE id={user_id}"
  cur.execute(query)
  return cur.fetchone()

def process_users(ids):
  results = []
  for i in range(len(ids)):
    results.append(get_user(ids[i]))
  return results
